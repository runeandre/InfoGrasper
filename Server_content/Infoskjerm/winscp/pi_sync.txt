# Security how to
# Use these commands once to accept certificate: 
# C:\Infoskjerm\winscp\winscp.exe sftp://pi:raspberry@infoskjerm1/
# C:\Infoskjerm\winscp\winscp.exe sftp://pi:raspberry@infoskjerm2/
#
# Connect infoskjerm1
open sftp://pi:raspberry@infoskjerm1/
# Change remote directory
cd /home/pi/Pictures
# Force binary mode transfer
option transfer binary
# Syncronize
synchronize remote -delete -criteria=size C:\Infoskjerm\MediaFolders\Output /home/pi/Pictures
# Disconnect infoskjerm1
close
# Connect infoskjerm2
open sftp://pi:raspberry@infoskjerm2/
# Change remote directory
cd /home/pi/Pictures
# Force binary mode transfer
option transfer binary
# Syncronize
synchronize remote -delete -criteria=size C:\Infoskjerm\MediaFolders\Output /home/pi/Pictures
# Disconnect infoskjerm2
close
# Exit WinSCP
exit
